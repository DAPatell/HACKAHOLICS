# Email Configuration Example
# Copy these settings to your environment variables or app config

# Gmail SMTP Settings
MAIL_USERNAME=your-email@gmail.com
MAIL_PASSWORD=your-app-password
MAIL_DEFAULT_SENDER=noreply@expensemanager.com

# Frontend URL for password reset links
FRONTEND_URL=http://localhost:3000

# JWT Secret Key (change in production)
JWT_SECRET_KEY=your-secret-key-change-this-in-production

# Instructions:
# 1. For Gmail, use an App Password instead of your regular password
# 2. Enable 2-factor authentication on your Gmail account
# 3. Generate an App Password: Google Account > Security > App passwords
# 4. Use the generated app password as MAIL_PASSWORD
